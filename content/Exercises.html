Exercises

1. 使用 printf 函数输出了 a、b 和 c 的值到屏幕，其中格式化字符串指定了输出的格式。

#include <stdio.h>

void main()
{
float a = 0.5;
double b = 1.2;
int c = 3;

b = b + a + c; // 浮点数和整数之间的运算

/* 输出 a、b、c 到屏幕 */
printf("a = %3.1f, b = %3.1f, c = %d", a, b, c);
}


2.x 的值为字符 'a'，y 的值为字符 'a'（通过ASCII码转换得到），以及 y 的ASCII值为整数 97。

#include <stdio.h>

int main()
{
char x, y;
x = 'a';
y = (char)97;

/* 输出 x, y, 以及 y 的 ASCII 值 */
printf("x = %c, y = %c, y 的 ASCII 值 = %d", x, y, y);

return 0;
}

3.使用 printf 函数输出这三个变量的值。格式化字符串中 %d 用于输出整数，%ld 用于输出长整数。

#include <stdio.h>

int main()
{
int a = 64;
int b = 0x40; // 以十六进制表示
long c = 64L;

printf("%d, %d, %ld", a, b, c);

return 0;
}

4.這個程式實現了從使用者獲取一個整數輸入，並將其顯示出來。

#include <stdio.h>

int main()
{
int i;
printf("输入一个整数：");
scanf("%d", &i); /* 使用 &（地址运算符）获取变量 i 的地址 */
printf("输入的数字是 %d", i);

return 0;
}

5.這個程式演示了基本的數學運算，包括乘法、除法、加法、減法和求餘數。

#include <stdio.h>

int main()
{
int a, b;
a = 10; b = 3;

printf("%d \n", a * b); // 乘法
printf("%d \n", a / b); // 除法
printf("%d \n", a + b); // 加法
printf("%d \n", a - b); // 减法
printf("%d \n", a % b); // 求余

return 0;
}

6.這個程式演示了比較運算符號的使用，並輸出了不同的比較結果，這些結果以整數形式表示，1 代表 True，0 代表 False。

#include <stdio.h>

int main()
{
int a = 10, b = 5;

printf("a == b is %d \n", a == b); // 等于
printf("a > b is %d \n", a > b); // 大于
printf("a < b is %d \n", a < b); // 小于
printf("a >= b is %d \n", a >= b); // 大于等于
printf("a <= b is %d \n", a <= b); // 小于等于
printf("a != b is %d \n", a != b); // 不等于

printf("\n");

b = 10;
printf("a == b is %d \n", a == b); // 等于
printf("a > b is %d \n", a > b); // 大于
printf("a < b is %d \n", a < b); // 小于
printf("a >= b is %d \n", a >= b); // 大于等于
printf("a <= b is %d \n", a <= b); // 小于等于
printf("a != b is %d \n", a != b); // 不等于

return 0;
}

7.這個程式演示了不同的位元運算符號的使用，包括按位或、按位與、按位異或、位元左移、位元右移、以及按位取反（補碼運算）。

#include <stdio.h>

void main()
{
int a, b;
a = 15;
b = 1;

printf("%d \n", a | b); /* a OR b（按位或运算） */
printf("%d \n", a & b); /* a AND b（按位与运算） */
printf("%d \n", a ^ b); /* a XOR b（按位异或运算） */
printf("%d \n", a << 1); /* a 位元左移 1 位 */
printf("%d \n", a >> 1); /* a 位元右移一位 */
printf("%d \n", ~a); /* a 的补码运算 */

return 0;
}

8.

這個程式演示了邏輯非運算符號 ! 的使用，它將非零值取反為 0，將零值取反為 1。

#include <stdio.h>

void main()
{
int a;
a = 3;
printf("%d\n", !a); /* 打印 a 的逻辑非结果（a 不为 0，则逻辑非为 0） */

a = 0;
printf("%d\n", !a); /* 打印 a 的逻辑非结果（a 为 0，则逻辑非为 1） */

return 0;
}

9.這個程式演示了邏輯與運算符號 && 的使用，它在兩個表達式都為真時返回真，只要有一個表達式為假就返回假

#include <stdio.h>

void main()
{
printf("%d\n", 1 && 3); /* 打印两个表达式的逻辑与结果（两者都为真，则结果为真） */
printf("%d\n", 0 && 0); /* 打印两个表达式的逻辑与结果（只要有一个为假，则结果为假） */
printf("%d\n", 2 && 2); /* 打印两个表达式的逻辑与结果（两者都为真，则结果为真） */

return 0;
}

char a;

printf("char 类型变量 a 的大小为 %d 字节\n", sizeof(a));

return 0;
}
