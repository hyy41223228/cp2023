1.，這個程式接收使用者輸入的一個字符，然後將該字符顯示出來。使用 getchar() 來讀取輸入的字符，並使用 putchar() 來輸出該字符。這個程式特別適合用來演示基本的字符輸入和輸出。

#include <stdio.h>

/* 读取和显示一个字符 */
int main() {
char c;

printf("请输入一个字符： ");
c = getchar(); /* 读取输入的字符 */

printf("输入的字符是： ");
putchar(c); /* 输出字符 */

return 0;
}

2.這個程式接收使用者輸入的兩個整數，計算它們的總和，然後顯示結果。使用 scanf 來讀取輸入的整數，並使用 printf 來顯示結果。

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

/* 读取两个整数，计算它们的总和并打印结果 */
int main()
{
int this_is_a_number1, this_is_a_number2, total;

printf("请输入一个整数：\n ");
scanf("%d", &this_is_a_number1); /* 读取输入的整数 */
printf("你输入了 %d\n", this_is_a_number1);

printf("请输入另一个整数：\n ");
scanf("%d", &this_is_a_number2); /* 读取输入的整数 */
printf("你输入了 %d\n", this_is_a_number2);

total = this_is_a_number1 + this_is_a_number2; /* 计算两个整数的总和 */
printf("总和为 %d\n", total);

return 0;
}

6.這個程式使用 for 循環計算了使用者輸入的十個浮點數的總和。程式通過 scanf 函式讀取輸入，並使用 printf 函式顯示計算結果。

##define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

/* 演示 for 循环：计算用户输入的十个浮点数的总和 */
int main()
{
float this_is_a_number, total;
int i;

total = 0;

/* for 循环执行 10 次 */
for (i = 0; i < 10; i++)
{
printf("请输入一个数字：\n");
scanf("%f", &this_is_a_number); /* 读取输入的数字 */
total = total + this_is_a_number;
}

printf("总和为：%f\n", total);

return 0;
}


7.這個程式演示了嵌套的 for 循環，計算了使用者輸入的二維數組（10行2列）的總和。程式通過 scanf 函式讀取輸入，並使用 printf 函式顯示計算結果。

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

/* 演示嵌套的 for 循环：计算用户输入的二维数组（10行2列）的总和 */
int main()
{
float this_is_a_number, total;
int i, j;

total = 0;

/* 外部 for 循环执行 10 次 */
for (i = 0; i < 10; i++)
{
/* 内部 for 循环执行 2 次 */
for (j = 0; j < 2; j++)
{
printf("请输入一个数字：\n");
scanf("%f", &this_is_a_number); /* 读取输入的数字 */
total = total + this_is_a_number;
}
}

printf("总和为：%f\n", total);

return 0;
}


8.這個程式演示了使用 do-while 循環計算使用者輸入的十個浮點數的總和。程式通過 scanf 函式讀取輸入，並使用 printf 函式顯示計算結果。 do-while 循環確保至少執行一次，即使條件在循環開始時為 false。

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

/* 演示 do-while 循环：计算用户输入的十个浮点数的总和 */
int main()
{
float this_is_a_number, total;
int i;

total = 0;
i = 0;

/* do-while 循环执行，直到 i 的值达到 10 */
do
{
printf("请输入一个数字：\n");
scanf("%f", &this_is_a_number); /* 读取输入的数字 */
total = total + this_is_a_number;
i++;
} while (i < 10);

printf("总和为：%f\n", total);

return 0;
}


9.這個程式使用 switch 语句根据使用者輸入的整數值執行不同的操作。根據 this_is_a_number 的值，程式會進入相應的 case 分支或 default 分支。

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

/* switch语句示例：根据用户输入的整数值执行不同的操作 */
int main()
{
int this_is_a_number;

printf("请输入一个介于1和5之间的整数：\n ");
scanf("%d", &this_is_a_number);

switch (this_is_a_number)
{
case 1:
printf("Case1: 值是：%d", this_is_a_number);
break;
case 2:
printf("Case2: 值是：%d", this_is_a_number);
break;
case 3:
printf("Case3: 值是：%d", this_is_a_number);
break;
case 4:
printf("Case4: 值是：%d", this_is_a_number);
break;
case 5:
printf("Case5: 值是：%d", this_is_a_number);
break;
default:
printf("错误：值是：%d", this_is_a_number); /* 输入的数字不在1和5之间 */
}

return 0;
}


10.這個程式使用 if 和 if-else 语句根據使用者輸入的整數值執行不同的條件分支。程式根據條件的滿足與否，輸出相應的信息。

#include <stdio.h>

/* if语句示例：根据用户输入的整数值执行不同的条件分支 */
int main()
{
int this_is_a_number;

printf("请输入一个介于1和10之间的整数：\n ");
scanf("%d", &this_is_a_number);

if (this_is_a_number < 6)
{
printf("这个数字小于6;\n ");
}

printf("请输入一个介于10和20之间的整数：\n ");
scanf("%d", &this_is_a_number);

if (this_is_a_number < 16)
{
printf("这个数字小于16\n ");
}
else
{
printf("这个数字大于等于16\n ");
}

return 0;
}
